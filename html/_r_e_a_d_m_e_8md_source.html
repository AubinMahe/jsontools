3 ## Features ##
4 
5 * Tree oriented: each node is strongly typed
6 * Dynamically allocated: like a DOM XML parser, this parser reads all the data in memory,
7  chunks allocated by `calloc` and `strdup`.
8 * Reading a file or a character buffer
9 * Getting a JST_Element by its path
10 * Creates, read, updates or deletes a property of an object or an item of an array.
11 * Documentation built with [Doxygen](https://www.doxygen.nl/index.html)

26 ## Sample code ##
27 
28 This sample code:
29 * read the json file ["data/object.json"](https://raw.githubusercontent.com/AubinMahe/jsontools/master/data/object.json)
30 * search an element in the loaded tree via the path `"repository.examples[2].subs[3].value"`
31 * print the value found
32 * serialize as text the tree and print it
33 * free the allocated resources
34 
35 ```
36 #include "jstools.h"
37 
38 #include <stdio.h>
39 #include <stdlib.h>
40 
41 int main( void ) {
42  const char * json_filename = "data/object.json";
43  JST_Element root = JST_Element_Zero;
44  JST_SyntaxError syntax_error = JST_SyntaxError_Zero;
45  JST_Error err = JST_load_from_file( json_filename, &root, &syntax_error );
46  if( err != JST_ERR_NONE ) {
47  return err;
48  }
49  const char * json_path = "repository.examples[2].subs[3].value";
50  JST_Element * elt = NULL;
51  if( ( JST_get( json_path, &root, &elt ) == JST_ERR_NONE )
52  &&( elt != NULL )
53  &&( elt->type == JST_INTEGER ))
54  {
55  printf( "%s = %ld\n", json_path, elt->value.integer );
56  }
57  char * json_text = NULL;
58  if( JST_serialize( &root, &json_text, 2 ) == JST_ERR_NONE ) {
59  printf( "%s\n", json_text );
60  free( json_text );
61  }
62  JST_delete_element( &root );
63  return 0;
64 }
